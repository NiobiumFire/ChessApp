    Handles HTTP

------ to install ------

backend:

# 1. Go to backend folder
cd backend

# 2. Create the virtual environment
python -m venv .venv

# 3. Activate the venv (PowerShell)
Set-ExecutionPolicy -Scope Process -ExecutionPolicy Bypass
.\.venv\Scripts\Activate.ps1

# 4. Install required Python packages in the venv
pip install -r requirements.txt

frontend:

# 1. Go to frontend folder
cd frontend

# 2. Create Vite React+TypeScript project
Set-ExecutionPolicy -Scope Process -ExecutionPolicy Bypass
npm create vite@latest
    name:
    React
    TypeScript

# 3. Go to frontend *project* folder
cd frontend\chess-frontend

# 3. Install chess-related packages
npm install react-chessboard chess.js @types/chess.js

# 4. Test suite (Vitest)
cd frontend\chess-frontend
Set-ExecutionPolicy -Scope Process -ExecutionPolicy Bypass
npm install --save-dev vitest @vitest/ui jsdom
npm install --save-dev vite-tsconfig-paths
npm install --save-dev @testing-library/react @testing-library/user-event
npm install --save-dev @testing-library/jest-dom @types/testing-library__jest-dom

# 5. For testing on commit and preventing commit on fail, from the <root> directory
npm install -D husky
npx husky init

# 6. Set VITE_CHESSAPP_BACKEND_URL environment variable

------ to run ------

backend:

new terminal

# 1. Go to backend folder
# 2. Activate venv
# 3. Run backend server

cd backend
Set-ExecutionPolicy -Scope Process -ExecutionPolicy Bypass
.\.venv\Scripts\Activate.ps1
uvicorn main:app --reload

frontend:

new terminal

# 1. Go to frontend *project* folder
# 2. Start frontend dev server

cd frontend/chess-frontend
Set-ExecutionPolicy -Scope Process -ExecutionPolicy Bypass
npm run dev

------ to test ------

backend:

cd backend
Set-ExecutionPolicy -Scope Process -ExecutionPolicy Bypass
.\.venv\Scripts\Activate.ps1
pytest

or

cd backend
./.venv/Scripts/python.exe -m pytest

frontend:

cd frontend/chess-frontend
Set-ExecutionPolicy -Scope Process -ExecutionPolicy Bypass
npm test

npm run test:ui

Test the tests:
from project root
git commit --allow-empty -m "test"
